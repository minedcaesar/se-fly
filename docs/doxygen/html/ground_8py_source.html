79@bp.route('/move-aircraft', methods=['GET', 'POST'])
80@role_required('ground_op_manager')
81@reauth_required
82def move_aircraft():
83 db = get_db()
84 if request.method == 'POST':
85 aircraft_id = request.form.get('aircraft_id')
86 target_gate = request.form.get('target_gate', '').strip()
87 reason = request.form.get('reason', '').strip()
88 ac = db.execute('SELECT * FROM aircraft WHERE id = ?', (aircraft_id,)).fetchone()
89 if ac is None:
90 flash('Aircraft not found.', 'danger')
91 return redirect(url_for('ground.move_aircraft'))
92 db.execute('UPDATE aircraft SET current_position = ? WHERE id = ?',
93 (target_gate, aircraft_id))
94 db.execute(
95 'INSERT INTO accountability_log_entries '
96 '(timestamp, manager_id, staff_id, reason_for_change) VALUES (?, ?, ?, ?)',
97 (datetime.now().isoformat(), session['user_id'], session['user_id'],
98 f'Moved {ac["registration"]} to {target_gate}: {reason}'),
99 )
100 db.commit()
101 session.pop('reauthed', None) # consume the re-auth
102 flash('Aircraft moved.', 'success')
103 return redirect(url_for('ground.move_aircraft'))
104 aircraft = db.execute('SELECT * FROM aircraft ORDER BY registration').fetchall()
105 return render_template('ground/move_aircraft.html', aircraft=aircraft)
